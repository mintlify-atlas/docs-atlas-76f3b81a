## Usage Examples

<CodeGroup>
```typescript Graph Operations
import {
  addEdge,
  getIncomers,
  getOutgoers,
  getConnectedEdges,
  connectionExists
} from '@vue-flow/core'

const nodes = ref([...])
const edges = ref([...])

// Add a new edge
const newConnection = {
  source: 'node-1',
  target: 'node-2',
  sourceHandle: 'output-1',
  targetHandle: 'input-1'
}

if (!connectionExists(newConnection, edges.value)) {
  edges.value = addEdge(newConnection, edges.value)
}

// Get related nodes
const incoming = getIncomers('node-2', nodes.value, edges.value)
const outgoing = getOutgoers('node-2', nodes.value, edges.value)
const connected = getConnectedEdges(['node-2'], edges.value)
```

```typescript Viewport Calculations
import {
  getTransformForBounds,
  getRectOfNodes,
  getNodesInside,
  pointToRendererPoint
} from '@vue-flow/core'
import { useVueFlow } from '@vue-flow/core'

const { nodes, viewport, setViewport } = useVueFlow()

// Fit selected nodes in view
const selectedNodes = nodes.value.filter(n => n.selected)
const bounds = getRectOfNodes(selectedNodes)

const transform = getTransformForBounds(
  bounds,
  1200, // viewport width
  800,  // viewport height
  0.5,  // min zoom
  2,    // max zoom
  0.1   // padding
)

setViewport(transform)

// Find nodes in selection area
const selectionRect = { x: 0, y: 0, width: 200, height: 200 }
const nodesInside = getNodesInside(
  nodes.value,
  selectionRect,
  viewport.value,
  true // partially
)
```

```typescript Coordinate Transformations
import {
  pointToRendererPoint,
  rendererPointToPoint
} from '@vue-flow/core'
import { useVueFlow } from '@vue-flow/core'

const { viewport } = useVueFlow()

// Convert mouse position to flow coordinates
function onMouseClick(event: MouseEvent) {
  const flowPosition = pointToRendererPoint(
    { x: event.clientX, y: event.clientY },
    viewport.value,
    false, // snapToGrid
    [15, 15] // snapGrid
  )
  
  console.log('Flow position:', flowPosition)
}

// Convert flow position to screen coordinates
const nodeScreenPos = rendererPointToPoint(
  { x: 100, y: 100 },
  viewport.value
)
```

```typescript Edge Path Utilities
import {
  getBezierPath,
  getSmoothStepPath,
  getStraightPath,
  Position
} from '@vue-flow/core'

// Calculate bezier path
const [path, labelX, labelY, offsetX, offsetY] = getBezierPath({
  sourceX: 0,
  sourceY: 0,
  sourcePosition: Position.Right,
  targetX: 200,
  targetY: 100,
  targetPosition: Position.Left,
  curvature: 0.25
})

// Use in custom edge component
const edgePath = computed(() => {
  return getBezierPath({
    sourceX: sourceX.value,
    sourceY: sourceY.value,
    sourcePosition: sourcePosition.value,
    targetX: targetX.value,
    targetY: targetY.value,
    targetPosition: targetPosition.value
  })
})
```

```typescript Type Guards
import {
  isNode,
  isEdge,
  isGraphNode,
  isGraphEdge
} from '@vue-flow/core'

function processElement(element: Node | Edge) {
  if (isNode(element)) {
    console.log('Node position:', element.position)
  } else if (isEdge(element)) {
    console.log('Edge connection:', element.source, '->', element.target)
  }
}

function processGraphElement(element: GraphNode | GraphEdge) {
  if (isGraphNode(element)) {
    console.log('Computed position:', element.computedPosition)
    console.log('Dimensions:', element.dimensions)
  } else if (isGraphEdge(element)) {
    console.log('Source node:', element.sourceNode)
    console.log('Target node:', element.targetNode)
  }
}
```

## Function Signatures

### Graph Operations

```typescript
// Add edge to elements
function addEdge(
  edgeParams: Edge | Connection,
  elements: Elements,
  defaults?: DefaultEdgeOptions
): Elements

// Update edge connection
function updateEdge(
  oldEdge: Edge,
  newConnection: Connection,
  elements: Elements
): Elements

// Get connected edges
function getConnectedEdges<E extends Edge>(
  nodesOrId: Node[] | string,
  edges: E[]
): E[]

// Check connection existence
function connectionExists(
  edge: Edge | Connection,
  elements: Elements
): boolean
```

### Node Relationships

```typescript
// Get incoming nodes
function getIncomers<N extends Node>(
  nodeOrId: Node | { id: string } | string,
  nodes: N[],
  edges: Edge[]
): N[]

// Get outgoing nodes
function getOutgoers<N extends Node>(
  nodeOrId: Node | { id: string } | string,
  nodes: N[],
  edges: Edge[]
): N[]
```

### Viewport Calculations

```typescript
// Get transform for bounds
function getTransformForBounds(

// Get rectangle of nodes
function getRectOfNodes(nodes: GraphNode[]): Rect

function getNodesInside(

function getBoundsofRects(rect1: Rect, rect2: Rect): Rect
```

<Note>
  **Deprecated Utilities**: Some utilities like `addEdge`, `updateEdge`, and change application functions are deprecated for direct use. It's recommended to use the store instance methods from `useVueFlow()` instead:
  
  ```typescript
  const { addEdges, updateEdge, applyNodeChanges } = useVueFlow()
  ```
</Note>
